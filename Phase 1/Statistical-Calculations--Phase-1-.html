# Define the function to find the statistical mode
get_mode <- function(x) {
  uniq_vals <- unique(x)
  counts <- tabulate(match(x, uniq_vals))
  if (length(unique(counts)) == 1) {
    return("No mode")  # Return "No mode" if all values have the same frequency
  }
  uniq_vals[which.max(counts)]
}
